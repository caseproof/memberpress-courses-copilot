    initializeUIComponents: typeof window.initializeUIComponents,
    createNewConversation: typeof window.createNewConversation,
    showSessionManager: typeof window.showSessionManager
});

// Check send button handlers
var sendBtn = jQuery('#mpcc-send-message')[0];
if (sendBtn) {
    console.log('Send button direct events:', jQuery._data(sendBtn, 'events'));
}

// Check document-level handlers
var docEvents = jQuery._data(document, 'events');
if (docEvents && docEvents.click) {
    console.log('Document click handlers for #mpcc-send-message:');
    docEvents.click.forEach(function(handler, i) {
        if (handler.selector && handler.selector.includes('mpcc-send-message')) {
            console.log('  Handler', i, '- selector:', handler.selector);
        }
    });
}
